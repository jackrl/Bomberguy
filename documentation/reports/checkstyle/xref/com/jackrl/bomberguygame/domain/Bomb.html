9   /**
10   * Class that represents the bombs used by the player to destroy blocks and kill
11   * enemies
12   */
13  public class Bomb extends Entity {
14      
15      private Random rand = new Random();
16      
17      private Player player;
18      
19      private boolean isActive;
20      private boolean hasExploded;
21      private int timer;
22      private int explosionTime;
23      
24      protected ArrayList<Explosion> explosions;
25      
26      private int range;
27  
28      /**
29       * Constructor of the Bomb class.
30       * 
31       * @param  player
32       * @throws SlickException 
33       */
34      public Bomb(Player player) throws SlickException {
35          super(-100, -100);
36          this.player = player;
37          this.range = player.range;
38          resetBomb();
39          
40          sprite = new Image("rsc/sprites/bombSprite.png");
41      }
42      
43      private void resetBomb() {
44          isActive = true;
45          hasExploded = false;
46          timer = 3 * 1000;
47          explosionTime = 1 * 1000;
48          explosions = new ArrayList<Explosion>();
49      }
50  
51      /**
52       * Method that sets the location of the bomb based on the position of the
53       * player that owns it
54       */
55      public void setLocation() {
56          x = ((int) ((player.getX() + 16) / 32)) * 32; 
57          y = ((int) ((player.getY() + 16) / 32)) * 32;
58      }
59      
60      /**
61       * Method that updates the timer of the bomb until it goes bellow zero. When
62       * the timer goes bellow zero the bomb is deactivated.
63       * @param   delta   Time to advance
64       * @param   level   The level object containing all the other entities
65       * @throws  SlickException
66       */
67      public void update(int delta, Level level) throws SlickException {
68          timer -= delta;
69          if (hasExploded) {
70              explosionTime -= delta;
71              if (explosionTime <= 0)
72                  isActive = false;
73          } else if (timer <= 0) 
74              this.explode(level);
75          
76      }
77      
78      /**
79       * Method that controls the explosion of the bomb. 
80       * 
81       * @param  level
82       * @throws SlickException 
83       */
84      protected void explode(Level level) throws SlickException {
85          hasExploded = true;
86          
87          explosions.add(new Explosion(this.x, this.y));
88          
89          ArrayList<Integer[]> posXCoords = new ArrayList<Integer[]>();
90          ArrayList<Integer[]> negXCoords = new ArrayList<Integer[]>();
91          ArrayList<Integer[]> posYCoords = new ArrayList<Integer[]>();
92          ArrayList<Integer[]> negYCoords = new ArrayList<Integer[]>();
93          for (int i = 1; i <= range; i++) {
94              posXCoords.add(new Integer[]{(int) this.x + i * 32, (int) this.y});
95              negXCoords.add(new Integer[]{(int) this.x - i * 32, (int) this.y});
96              posYCoords.add(new Integer[]{(int) this.x, (int) this.y + i * 32});
97              negYCoords.add(new Integer[]{(int) this.x, (int) this.y - i * 32});
98          }
99          
100         checkExplosionsWithWalls(posXCoords, negXCoords, posYCoords, negYCoords, level.walls);
101         checkExplosionsWithBlocks(posXCoords, negXCoords, posYCoords, negYCoords, level.blocks, level.powerUps);
102         checkExplosionsWithBombs(posXCoords, negXCoords, posYCoords, negYCoords, level);
103         
104         for (Integer[] coord : posXCoords)
105             explosions.add(new Explosion(coord[0], coord[1]));
106         for (Integer[] coord : negXCoords)
107             explosions.add(new Explosion(coord[0], coord[1]));
108         for (Integer[] coord : posYCoords)
109             explosions.add(new Explosion(coord[0], coord[1]));
110         for (Integer[] coord : negYCoords)
111             explosions.add(new Explosion(coord[0], coord[1]));
112     }
113     
114     private void checkExplosionsWithWalls(ArrayList<Integer[]> posXCoords, ArrayList<Integer[]> negXCoords,
115                                           ArrayList<Integer[]> posYCoords, ArrayList<Integer[]> negYCoords,
116                                           ArrayList<Wall> walls) {
117         checkExplosionsWithWallsHelper(posXCoords, walls);
118         checkExplosionsWithWallsHelper(negXCoords, walls);
119         checkExplosionsWithWallsHelper(posYCoords, walls);
120         checkExplosionsWithWallsHelper(negYCoords, walls);
121     }
122     
123     private void checkExplosionsWithWallsHelper(ArrayList<Integer[]> coords, ArrayList<Wall> walls) {
124         boolean removeRest = false;
125         for (Iterator<Integer[]> iter = coords.iterator(); iter.hasNext();) {
126             Integer[] coord = iter.next();
127             
128             if (removeRest) {
129                 iter.remove();
130                 continue;
131             }
132             
133             for (Wall wall : walls) {
134                 if (wall.x == coord[0] && wall.y == coord[1]) {
135                     iter.remove();
136                     removeRest = true;
137                 }
138             }
139         }
140     }
141     
142     private void checkExplosionsWithBlocks(ArrayList<Integer[]> posXCoords, ArrayList<Integer[]> negXCoords,
143                                            ArrayList<Integer[]> posYCoords, ArrayList<Integer[]> negYCoords,
144                                            ArrayList<Block> blocks,
145                                            ArrayList<PowerUp> powerUps) throws SlickException {
146         checkExplosionsWithBlocksHelper(posXCoords, blocks, powerUps);
147         checkExplosionsWithBlocksHelper(negXCoords, blocks, powerUps);
148         checkExplosionsWithBlocksHelper(posYCoords, blocks, powerUps);
149         checkExplosionsWithBlocksHelper(negYCoords, blocks, powerUps);
150     }
151     
152     private void checkExplosionsWithBlocksHelper(ArrayList<Integer[]> coords, ArrayList<Block> blocks, ArrayList<PowerUp> powerUps) throws SlickException {
153         boolean removeRest = false;
154         for (Iterator<Integer[]> iter = coords.iterator(); iter.hasNext();) {
155             Integer[] coord = iter.next();
156             
157             if (removeRest) {
158                 iter.remove();
159                 continue;
160             }
161             
162             for (Iterator<Block> blockIter = blocks.iterator(); blockIter.hasNext();) {
163                 Block block = blockIter.next();
164                 
165                 if (block.x == coord[0] && block.y == coord[1]) {
166                     blockIter.remove();
167                     removeRest = true;
168                     
169                     switch (rand.nextInt(10)) {
170                         case 0:
171                             powerUps.add(new BombPowerUp(block.x, block.y));
172                             break;
173                         case 9:
174                             powerUps.add(new ExplosionPowerUp(block.x, block.y));
175                             break;
176                     }
177                     break;
178                 }
179             }
180         }
181     }
182     
183     private void checkExplosionsWithBombs(ArrayList<Integer[]> posXCoords, ArrayList<Integer[]> negXCoords,
184                                           ArrayList<Integer[]> posYCoords, ArrayList<Integer[]> negYCoords,
185                                           Level level) throws SlickException {
186         checkExplosionsWithBombsHelper(posXCoords, level.bombs, level);
187         checkExplosionsWithBombsHelper(negXCoords, level.bombs, level);
188         checkExplosionsWithBombsHelper(posYCoords, level.bombs, level);
189         checkExplosionsWithBombsHelper(negYCoords, level.bombs, level);
190     }
191     
192     private void checkExplosionsWithBombsHelper(ArrayList<Integer[]> coords, ArrayList<Bomb> bombs, Level level) throws SlickException {
193         boolean removeRest = false;
194         for (Iterator<Integer[]> iter = coords.iterator(); iter.hasNext();) {
195             Integer[] coord = iter.next();
196             
197             if (removeRest) {
198                 iter.remove();
199                 continue;
200             }
201             
202             for (Bomb bomb : bombs) {
203                 if (bomb.x == coord[0] && bomb.y == coord[1]) {
204                     iter.remove();
205                     if (!bomb.hasExploded)
206                         bomb.explode(level);
207                     removeRest = true;
208                     break;
209                 }
210             }
211         }
212     }

214     @Override
215     public void render() {
216         if (!hasExploded) {
217             super.render();
218         } else {
219             for (Explosion explosion : explosions) 
220                 explosion.render();
221         }

253     public boolean equals(Object obj) {
254         if (obj == null || obj.getClass() != this.getClass())
255             return false;
256         
257         Bomb other = (Bomb) obj;
258         
259         if (this.x == other.x && this.y == other.y)
260             return true;
261         
262         return false;
263     }
