1   package com.jackrl.bomberguygame.domain;
2   
3   import org.newdawn.slick.Image;
4   import org.newdawn.slick.SlickException;
5   import org.newdawn.slick.geom.Rectangle;
6   
7   /**
8    * <p>An abstract class for all the entities that are rendered in the game.</p>
9    */
10  public abstract class Entity {
11      
12      protected float x;
13      protected float y;
14      protected float maxSpeed = 0.f;
15      protected int dx;
16      protected int dy;
17      
18      protected Image sprite;
19      
20      private Rectangle boundingShape;
21      private Rectangle collidingShape;
22  
23      /**
24       * Constructor of the Entity class.
25       * 
26       * @param x
27       * @param y
28       * @throws SlickException 
29       */
30      public Entity(float x, float y) throws SlickException {
31          this.x = x;
32          this.y = y;
33          this.dx = 0;
34          this.dy = 0;
35          // A place holder in case that you forget to place the real one
36          sprite = new Image("rsc/sprites/placeHolderSprite.png");
37          
38          this.boundingShape = new Rectangle(x, y, sprite.getWidth() - 2, sprite.getHeight() - 2);
39          this.collidingShape = new Rectangle(-100, -100, 0, 0);
40      }
41  
42      // Might be used when coding the "AI" for aggressive enemies
43      public int getX() {
44          return (int) x;
45      }
46  
47      // Might be used when coding the "AI" for aggressive enemies
48      public int getY() {
49          return (int) y;
50      }
51      
52      /**
53       * Method that renders the sprite of an entity.
54       */
55      public void render() {
56          sprite.draw((int) x, (int) y);
57      }
58      
59      /**
60       * Method that checks the collisions between Entities
61       * 
62       * @param  other
63       * @return Whether this Entity has collided with the other Entity.
64       */
65      public boolean collidesWith(Entity other) {
66          boundingShape.setBounds((int) x, (int) y, sprite.getWidth() - 3, sprite.getHeight() - 3);
67          
68          collidingShape.setBounds((int) other.x, (int) other.y, other.sprite.getWidth() - 3, other.sprite.getHeight() - 3);
69          
70          return boundingShape.intersects(collidingShape);
71      }
72  }
