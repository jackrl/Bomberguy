1   package com.jackrl.bomberguygame.game.states;
2   
3   import com.jackrl.bomberguygame.util.Controller;
4   import org.newdawn.slick.GameContainer;
5   import org.newdawn.slick.Graphics;
6   import org.newdawn.slick.SlickException;
7   import org.newdawn.slick.state.BasicGameState;
8   import org.newdawn.slick.state.StateBasedGame;
9   
10  /**
11   * <p>A Slick2D state that represents the end screen of the game.</p>
12   */
13  public class EndState extends BasicGameState {
14      private Controller controller;
15      
16      @Override
17      public void init(GameContainer container, StateBasedGame game) throws SlickException {
18          controller = new Controller(container, game);
19      }
20  
21      @Override
22      public void render(GameContainer container, StateBasedGame game, Graphics g) throws SlickException {
23          g.drawString("Congratulations you have beaten the game!", 22, 160);
24          g.drawString("Press Space go to the start screen.", 55, 200);
25      }
26  
27      @Override
28      public void update(GameContainer container, StateBasedGame game, int delta) throws SlickException {
29          if (controller.controlInStartAndEndScreen())
30              game.enterState(State.START);
31      }
32      
33      @Override
34      public int getID() {
35          return State.END;
36      }
37  }
